#include <string>
#include <algorithm>
#include <vector>
#include <cmath>
#include <iostream>
#include <iterator>
#include <tuple>
#include <fstream>

using namespace std;

int main() {
	string input = "", dummy = "", actual_input = "";
	int low, high, counter = 0, result = 0, input_counter = 0;
	char required;
	size_t i;
	while (cin >> input) {
		if (input == "END") { break; }
		if (input_counter < 3) {
			++input_counter;
			actual_input += input;
			actual_input.push_back(' ');
			if (input_counter != 3) { continue; } 
		}
		i = 0;
		cout << "analysing: " << actual_input << "\n";
		while (actual_input[i] != '-') {
			dummy.push_back(actual_input[i]);
			++i;
		}
		++i;
		low = stoi(dummy);
		dummy = "";
		while (actual_input[i] != ' ') {
			dummy.push_back(actual_input[i]);
			++i;
		}
		high = stoi(dummy);
		dummy = "";
		++i;
		required = actual_input[i];
		i += 3;
		for (size_t j = i; j < actual_input.size(); ++j) {
			if (actual_input[j] == required) {
				++counter;
			}
		}
		if ((low <= counter) && (counter <= high)) {
			++result;
		}
		counter = 0;
		input_counter = 0;
		actual_input = "";
	}
	cout << "RESULT: " << result << "\n";
	return 0;
}

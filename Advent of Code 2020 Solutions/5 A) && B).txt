#include <string>
#include <algorithm>
#include <vector>
#include <cmath>
#include <iostream>
#include <iterator>
#include <tuple>

using namespace std;

int main() {
	string in = "", first = "", second = "";
	int row = 0, col = 0, dummy = 1, seat_id = 0;
	vector<int> IDs = {};
	while (cin >> in) {
		if (in == "END") { break; }
		for (int i = 6; i >= 0; --i) {
			if (in[i] == 'B') {
				for (int x = 0; x != 6 - i; ++x) {
					dummy *= 2;
				}
				row += dummy;
				dummy = 1;
			}
		}
		for (int i = 9; i >= 7; --i) {
			if (in[i] == 'R') {
				for (int x = 0; x != 9 - i; ++x) {
					dummy *= 2;
				}
				col += dummy;
				dummy = 1;
			}
		}
		IDs.push_back(((row * 8) + col));
		if (((row * 8) + col) > seat_id) {
			seat_id = ((row * 8) + col);
		}
		col = 0; row = 0;
	}
	vector<int> vec = {};
	for (int i = 8; i < 886; ++i) {
		vec.push_back(i);
	}
	for (size_t i = 0; i < IDs.size(); ++i) {
		for (size_t j = 0; j < vec.size(); ++j) {
			if (IDs[i] == vec[j]) {
				vec.erase(vec.begin() + j);
				break;
			}
		}
	}
	for (size_t i = 0; i < vec.size(); ++i) {
		cout << vec[i];
	}
	cout << "\n";
	cout << "ANSWERS:\n\tA) " << seat_id << "\n\tB) " << vec[0] << "\n";
	return 0;
}

#include <string>
#include <algorithm>
#include <vector>
#include <cmath>
#include <iostream>
#include <iterator>
#include <tuple>
#include <fstream>

using namespace std;

int main() {
	int n;
	bool found = false;
	vector<int>V;
	while (cin >> n) {
		V.push_back(n);
	}
	for (size_t i = 0; i < V.size(); ++i) {
		for (size_t j = i+1; j < V.size(); ++j) {
			if (V[i] + V[j] == 2020) {
				n = V[i] * V[j];
				found = true;
				break;
			}
		}
		if (found) { break; }
	}
	cout << n << "\n";
	return 0;
}

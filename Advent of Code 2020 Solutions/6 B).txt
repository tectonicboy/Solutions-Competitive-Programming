#include <string>
#include <algorithm>
#include <vector>
#include <cmath>
#include <iostream>
#include <iterator>
#include <tuple>

using namespace std;


int all_yeses(string &str, int x) {
	int c = 0, c_curr = 0;
	for (char chr = 97; chr < 123; ++chr) {
		c_curr = count(str.begin(), str.end(), chr);
		if (c_curr == x) { ++c; }
	}
	return c;
}

int main() {
	string line = "", all_answers = "";
	bool found = false;
	int count = 0, cur_res = 0, people = 0;
	while (getline(cin, line)) {
		if (line == "END") { 
			count += all_yeses(all_answers, people);
			break; 
		}

		//Resets may be required here.
		else if (line == "") { 
			count += all_yeses(all_answers, people);
			people = 0;
			all_answers = "";
			continue; 
		}

		else {
			++people;
			for (size_t i = 0; i < line.size(); ++i) {
				all_answers.push_back(line[i]);
			}
		}

	}
	cout << "Answer: " << count << "\n";
	return 0;
}
